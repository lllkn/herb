## 一、概述

第一章「翠竹村的春天」是百草行的首个正式章节，承接序章结业后的旅程。玩家从岐黄学堂出发，途经**城郊平原**，抵达**翠竹村**行医，最终深入**溪流山谷**发现蛊根草线索，为第二章埋下伏笔。

## 三、数据格式规范

### 3.1 `map_events.json` 事件字段

```jsonc
{
  "eventId": "EVT_PLAINS_ENTRY",   // 事件唯一ID（大写，EVT_ 前缀）
  "trigger": "zone",                   // 触发方式：zone / npc / object / condition
  "zoneId": "zone_map_entry",          // trigger=zone 时必填
  "npcId": "woodcutter_npc",        // trigger=npc 时必填
  "objectId": "abandoned_basket",   // trigger=object 时必填
  "once": true,                       // 是否只触发一次
  "dialogType": "overlay",             // 对话呈现：bubble（头顶）/ overlay（底部对话框）
  "locComplete": "loc_herb_garden",  // 填写后标记该地点完成
  "condition": "all_locations_complete", // trigger=condition 时必填
  "conditionHerbs": [...],           // trigger=condition + has_herbs 时必填
  "requiredLocations": [...],       // trigger=condition + all_locations_complete 时必填
  "_note": "桥接 IntroScene→GameScene", // 开发备注（不参与逻辑）
  "steps": [...]                      // 剧情步骤（IntroScene 格式）
}
```

### 3.2 `story_chapter1.json` 场景字段

```jsonc
{
  "id": "C01",                 // 场景ID（C + 两位数）
  "name": "出生区搭车过场", // 场景名称（用于调试）
  "type": "scene",             // 场景类型：scene / dialogue / cg
  "background": "bg_plains_entry.png",  // 背景图片 key（不含 .png）
  "bgm": "bgm_journey.mp3",   // 背景音乐（可选）
  "_trigger": {                  // 地图触发接口（IntroScene 忽略此字段）
    "type": "zone",
    "zoneId": "zone_map_entry",
    "once": true,
    "dialogType": "overlay",
    "eventId": "EVT_PLAINS_ENTRY"
  },
  "steps": [...]                // 步骤数组（同序章格式）
}
```

## 五、地图对接接口

`story_chapter1.json` 的每个场景包含 `_trigger` 字段，地图系统读取后自行注册触发逻辑。

### 5.1 对接示例（GameScene.js）

```javascript
// 地图系统读取 story_chapter1.json
const chapter1Data = this.cache.json.get('chapter1_data');

// 遍历场景，注册触发区
chapter1Data.scenes.forEach(scene => {
    const t = scene._trigger;
    if (!t) return;

    if (t.type === 'zone') {
        // 创建矩形触发区
        const zone = this.add.zone(t.zoneId, x, y, width, height);
        this.physics.add.existing(zone);
        zone.sceneId = scene.id;

        // 如果 once=true，检查是否已触发
        if (t.once && gameState.storyEventsFired.has(t.eventId)) {
            zone.destroy();
        }
    }

    if (t.type === 'npc') {
        // 找到对应 NPC，绑定 E 键交互
        const npc = this.npcs.get(t.npcId);
        if (npc) npc.sceneId = scene.id;
    }

    if (t.type === 'object') {
        const obj = this.objects.get(t.objectId);
        if (obj) obj.sceneId = scene.id;
    }
});

// 触发时调用
onZoneEnter(player, zone) {
    const sceneId = zone.sceneId;
    this.scene.start('IntroScene', { startSceneId: sceneId });
}

// IntroScene 修改 create() 接受 startSceneId
create(data) {
    if (data && data.startSceneId) {
        const idx = this.prologueData.scenes.findIndex(s => s.id === data.startSceneId);
        if (idx >= 0) this._startScene(idx);
    } else {
        this._startScene(0);
    }
}
```

### 5.2 触发条件类型

| `trigger` | 说明 | 对应字段 |
|-------------|------|------------|
| `zone` | 进入矩形区域自动触发 | `zoneId` |
| `npc` | 靠近 NPC 后按 E 键 | `npcId` |
| `object` | 靠近地物后按 E 键 | `objectId` |
| `condition` | 满足特定条件时自动触发 | `condition`（`all_locations_complete` / `herb_count`） |
